#!/bin/bash
# Script com validações de Tel,E-mail,CPF,IP,Data...
####################### NOME ################################################
echo 'Informe seu Nome. Obs; NOME COMPLETO';
read NOME
######################### VALIDAR E-MAIL ##############################################
echo 'Informe o E-mail';
read EMAIL
echo $EMAIL | egrep '^([a-zA-Z0-9_-.])+@[0-9a-zA-Z.-]+\.[a-z]{2,3}$' && echo -e '\033[01;32m E-mail válido! \033[0m' || echo -e '\033[01;31m NÃO é válido esse E-mail.\033[0m'
####################### VALIDAR TELEFONE CELULAR no formato: (99)99999-9999 ################################################
echo 'Informe o número de Telefone CELULAR. Formato: (99)99999-9999';
read TELEFONE
echo $TELEFONE | egrep '^[(][0-9]{2}[)][0-9]{5}+-[0-9]{4}$' && echo -e '\033[01;32m Número válido! \033[0m' || echo -e '\033[01;31m NÃO é válido esse número.\033[0m'
####################### VALIDAR TELEFONE FIXO no formato: (99)9999-9999 ################################################
echo 'Informe o número de Telefone FIXO. Formato: (99)9999-9999';
read TELEFONE2
echo $TELEFONE2 | egrep '^[(][0-9]{2}[)][0-9]{4}+-[0-9]{4}$' && echo -e '\033[01;32m Número válido! \033[0m' || echo -e '\033[01;31m NÃO é válido esse número.\033[0m'
######################### VALIDAR RG ##############################################
echo 'Informe o RG. Formato 99.999.999-9';
read RG
echo $RG | egrep '^[0-9]{2}[.][0-9]{3}[.][0-9]{3}[-][0-9]{1}$' && echo -e '\033[01;32m Número válido! \033[0m' || echo -e '\033[01;31m NÃO é válido esse número.\033[0m'
######################### VALIDAR CPF ##############################################
echo 'Informe o CPF. Formato 999999999/99';
read CPF
echo $CPF | egrep '^[0-9]{9}[/][0-9]{2}$' && echo -e '\033[01;32m Número válido! \033[0m' || echo -e '\033[01;31m NÃO é válido esse número.\033[0m'

######################### VALIDAR DATA formato dd/mm/aaaa ##############################################
echo 'Informe a Data Nascimento. Formato dd/mm/aaaa';
read DATA
echo $DATA | egrep '^[0-3]{1}[0-9]{1}[/][0-1]{1}[0-9]{1}[/][0-9]{4}$' && echo -e '\033[01;32m Data válida! \033[0m' || echo -e '\033[01;31m NÃO é válida essa Data.\033[0m'
######################### VALIDAR IP ##############################################
echo 'Informe o número de IP';
read IP
echo $IP | egrep '^[0-9]{1,3}[.]{1}[0-9]{1,3}[.]{1}[0-9]{1,3}[.]{1}[0-9]{1,3}$' && echo -e '\033[01;32m IP válido! \033[0m' || echo -e '\033[01;31m NÃO é válido esse IP.\033[0m'
######################### VALIDAR MASCARA ##############################################
echo 'Informe o número de Mascara';
read MASCARA
echo $MASCARA | egrep '^[0-9]{1,3}[.]{1}[0-9]{1,3}[.]{1}[0-9]{1,3}[.]{1}[0-9]{1,3}$' && echo -e '\033[01;32m Mascara válida! \033[0m' || echo -e '\033[01;31m NÃO é válida essa Mascara.\033[0m'
